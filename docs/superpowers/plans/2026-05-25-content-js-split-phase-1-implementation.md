**Goal:** Mechanically extract ~1,500 LOC of shared widget helpers from the 4,674-line `content.js` into focused files under `src/widget/`. No behavior change — `dist/content.js` is functionally identical to the previous build at every commit.

## File-extraction pattern

Every extracted file follows the same shape:

```js
'use strict';

<verbatim slice from content.js>

if (typeof module !== 'undefined' && module.exports) {
  module.exports = {
    // names of any top-level functions / vars the slice declares
  };
}
```

The CJS export footer makes the file Node-test-loadable. In the
browser the footer is a no-op (no `module` global); the bundled
`dist/content.js` strips it via the same regex the solver bundler
uses.

After extracting a slice, content.js gets that slice removed (Edit
with the slice as `old_string`, empty as `new_string`). The bundler
re-assembles everything.

The line numbers in this plan reference the original pre-refactor file.

- [ ] **Step 2: Write `scripts/build-content-bundle.js`**

```js
'use strict';
// Concatenate src/widget/*.js (in deterministic dependency order) + the
// remaining content.js into dist/content.js. Mirror of
// scripts/build-solver-bundle.js — the extension content script needs a
// single concatenated file because the browser loads it via
// manifest.json's content_scripts.js entry (no module system).

const fs = require('fs');
const path = require('path');

// Order is deliberate: state must precede anything that touches it,
// the registry must precede dispatchers, makeWidget last. New files go
// at the bottom of their dependency tier.
const WIDGET_FILES = [
  'state.js',
  'worker.js',
  'cache.js',
  'galaxies-hint.js',
  'hint.js',
  // Phase-2 additions (preview.js, widget.js) slot in here later.
];

// Strip the trailing CommonJS export block from a per-file source. The
// block is the LAST `if (typeof module ...) { module.exports = {...} }`
// in each file; anchor on the closing `}` of the file.
const EXPORT_RE =
  /\n\s*if\s*\(\s*typeof\s+module[\s\S]*?module\.exports\s*=\s*\{[\s\S]*?\}\s*;?\s*\}\s*$/;

const root = path.join(__dirname, '..');
const widgetDir = path.join(root, 'src', 'widget');
const parts = [
  "'use strict';",
  "// Generated by scripts/build-content-bundle.js — do not edit by hand.",
  "",
];

// Phase-1 src/widget/*.js — only emit if present (lets the bundler work
// before any extraction has happened, useful during Phase-1 task 1).
for (const file of WIDGET_FILES) {
  const fullPath = path.join(widgetDir, file);
  if (!fs.existsSync(fullPath)) continue;
  let body = fs.readFileSync(fullPath, 'utf8');
  body = body.replace(/^\s*'use strict';\s*\n/, '');
  body = body.replace(EXPORT_RE, '');
  parts.push(`// ── src/widget/${file} ──`);
  parts.push(body.trim());
  parts.push("");
}

// Then the remaining content.js (the makeWidget shell + everything not
// yet extracted).
const contentBody = fs.readFileSync(path.join(root, 'content.js'), 'utf8')
  // If content.js gets reduced to a shim later, strip the require-shim form
  // so we don't emit a `module.exports = require(...)` in the browser bundle.
  .replace(/^\s*'use strict';\s*module\.exports\s*=\s*require\([^)]*\)\s*;\s*$/m, '');
parts.push('// ── content.js ──');
parts.push(contentBody.trim());
parts.push('');

const outDir = path.join(root, 'dist');
fs.mkdirSync(outDir, { recursive: true });
fs.writeFileSync(path.join(outDir, 'content.js'), parts.join('\n'));
console.log('Wrote dist/content.js');
```

```js
function loadWidgetSources() {
  const ctx = makeBrowserContext();
  ctx.globalThis = ctx;
  ctx.self = ctx;
  vm.createContext(ctx);
  // Solver source files
  const solverDir = path.join(REPO, 'src', 'solvers');
  const solverFiles = fs.readdirSync(solverDir)
    .filter(f => f.endsWith('.js') && f !== 'index.js')
    .map(f => path.join(solverDir, f));
  for (const fullPath of solverFiles) {
    vm.runInContext(fs.readFileSync(fullPath, 'utf8'), ctx,
      { filename: path.basename(fullPath) });
  }
  // Widget source files (extracted by Phase 1 / Phase 2 of the
  // content.js split). Order matters: this matches the order the
  // bundler concatenates them in.
  const widgetDir = path.join(REPO, 'src', 'widget');
  const widgetOrder = ['state.js', 'worker.js', 'cache.js',
                        'galaxies-hint.js', 'hint.js'];
  for (const f of widgetOrder) {
    const fp = path.join(widgetDir, f);
    if (!fs.existsSync(fp)) continue;
    vm.runInContext(fs.readFileSync(fp, 'utf8'), ctx, { filename: f });
  }
  for (const file of ['handler.js', 'content.js']) {
    vm.runInContext(fs.readFileSync(path.join(REPO, file), 'utf8'), ctx,
      { filename: file });
  }
  return ctx;
}
```
